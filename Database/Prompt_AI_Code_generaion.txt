function PopulateSelect(selector, items) {

function LoadChildByParent(parentId, targetSelector) {
    $.get('/Controller/GetChildren', { parentId: parentId || '' }, function (data) {
        var $target = $(targetSelector).empty()
                        .append('<option value="">Select</option>');
        data.forEach(function (item) {
            $target.append(
                $('<option>', { value: item.value, text: item.text })
                    .attr('data-country', item.country || '')
                    // add more data-* for other extra fields
            );
        });
        $target.trigger('change.select2');
    });
}

function EditRecord(id) {
    ShowLoader(true);
    $.get('/Controller/GetById', { id: id }, function (res) {
        ShowLoader(false);
        if (!res.success) { ShowToast(res.message, 'error'); return; }
        var d = res.data;
        ResetForm();
        $('#modalTitleText').text('Edit Record');

        // Set simple fields
        $('#recordId').val(d.recordId);
        $('#fieldOne').val(d.fieldOne);

        // For cascading dropdown — load children first, then set value,
        // then trigger change so auto-fill fires, with fallback
        LoadChildByParent(d.parentId, '#childId');
        setTimeout(function () {
            SetSelect2('#parentId', d.parentId);
            SetSelect2('#childId', d.childId);
            $('#childId').trigger('change');  // fires auto-fill via change event
            // Fallback in case data-* not yet on option
            if (!$('#autoFilledField').val()) {
                $('#autoFilledField').val(d.autoFilledValue);
            }
        }, 450);

        $('#myModal').modal('show');
    }).fail(function () {
        ShowLoader(false);
        ShowToast('Failed to fetch record.', 'error');
    });
}

function SaveRecord() {
    if (!ValidateForm()) return;

    var payload = {
        recordId:   parseInt($('#recordId').val()) || 0,
        fieldOne:   $('#fieldOne').val().trim(),
        fieldTwo:   $('#fieldTwo').val(),
        parentId:   parseInt($('#parentId').val()),
        childId:    parseInt($('#childId').val()),
        startDate:  $('#startDate').val(),
        endDate:    $('#endDate').val()
    };

    ShowLoader(true);
    $.ajax({
        url: '/Controller/Save',
        type: 'POST',
        contentType: 'application/json',
        data: JSON.stringify(payload),
        headers: { 'RequestVerificationToken': csrfToken },
        success: function (res) {
            ShowLoader(false);
            if (res.success) {
                $('#myModal').modal('hide');
                ShowToast(res.message, 'success');
                LoadGrid();
            } else {
                ShowToast(res.message, 'error');
            }
        },
        error: function () {
            ShowLoader(false);
            ShowToast('An unexpected error occurred.', 'error');
        }
    });
}

function DeleteRecord(id) {
    Swal.fire({
        title: 'Are you sure?',
        text: 'This record will be deleted.',
        icon: 'warning',
        showCancelButton: true,
        confirmButtonColor: '#d33',
        cancelButtonColor: '#6c757d',
        confirmButtonText: 'Yes, delete it!'
    }).then(function (result) {
        if (result.isConfirmed) {
            ShowLoader(true);
            $.ajax({
                url: '/Controller/Delete',
                type: 'POST',
                contentType: 'application/json',
                data: JSON.stringify(id),
                headers: { 'RequestVerificationToken': csrfToken },
                success: function (res) {
                    ShowLoader(false);
                    ShowToast(res.message, res.success ? 'success' : 'error');
                    if (res.success) LoadGrid();
                },
                error: function () {
                    ShowLoader(false);
                    ShowToast('Delete failed.', 'error');
                }
            });
        }
    });
}

function ValidateForm() {
    var msg = [];
    if (!$('#fieldOne').val().trim()) msg.push('Field One is required.');
    if (!$('#parentId').val())        msg.push('Parent is required.');
    if (!$('#childId').val())         msg.push('Child is required.');
    if (!$('#startDate').val())       msg.push('Start Date is required.');
    if (!$('#endDate').val())         msg.push('End Date is required.');

    if ($('#startDate').val() && $('#endDate').val()) {
        if (new Date($('#endDate').val()) <= new Date($('#startDate').val()))
            msg.push('End Date must be after Start Date.');
    }

    if (msg.length > 0) {
        Swal.fire({
            icon: 'warning',
            title: 'Validation Error',
            html: msg.map(function(m) { return '• ' + m; }).join('<br/>')
        });
        return false;
    }
    return true;
}

function FormatDate(dateStr) {
    if (!dateStr) return '';
    var d = new Date(dateStr);
    return String(d.getDate()).padStart(2,'0') + '-' +
           String(d.getMonth()+1).padStart(2,'0') + '-' +
           d.getFullYear();
}

function FormatDateInput(dateStr) {
    return dateStr ? new Date(dateStr).toISOString().split('T')[0] : '';
}

═══════════════════════════════════════════

C# Standards:

SQL Standards:

- Use GETDATE() for datetime defaults
- Never use SELECT * in stored procedures
- Always SET NOCOUNT ON at top of every SP
- Use WITH (NOLOCK) on all SELECT queries

jQuery / JS Standards:

- Send CSRF token in every POST AJAX header
- Use SweetAlert2 for all confirmations and toasts — never browser alert() or confirm()
- Use data-* attributes on option elements to carry extra data for auto-fill

Security:
- Validate all inputs server-side

- [ValidateAntiForgeryToken] on every POST action
- Sanitize output to prevent XSS

COMMON MISTAKES — NEVER DO THESE
═══════════════════════════════════════════

MISTAKE 1 — Missing [ValidateNever] on display fields
BAD:  public string PlantName { get; set; }
      → causes "The PlantName field is required" error on Save
GOOD: [ValidateNever] public string PlantName { get; set; }

MISTAKE 2 — Subclassing SelectListItem
BAD:  public class PlantItem : SelectListItem { public string Country { get; set; } }
      → causes "Cannot convert List<PlantItem> to List<SelectListItem>"
GOOD: Use Dictionary<string, string> for extra data per item

MISTAKE 3 — Cascading endpoint not returning extra fields
BAD:  return Json(list.Select(x => new { value = x.Value, text = x.Text }));
GOOD: return Json(list.Select(x => new { value = x.Value, text = x.Text, country = x.Country }));

MISTAKE 4 — Auto-fill not triggered in Edit mode
BAD:  SetSelect2('#plantId', d.plantId);
GOOD: LoadChildByParent(d.groupCompanyId, '#plantId');
      setTimeout(function() {
          SetSelect2('#plantId', d.plantId);
          $('#plantId').trigger('change');

MISTAKE 5 — Select2 z-index issue inside modal
BAD:  $('.select2-modal').select2({ placeholder: 'Select' });
GOOD: $('.select2-modal').select2({ dropdownParent: $('#myModal'), placeholder: 'Select' });

MISTAKE 6 — Auto-filled field not cleared on ResetForm
